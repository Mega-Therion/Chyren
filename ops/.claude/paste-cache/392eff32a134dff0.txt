import React, { useState, useEffect, useRef } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, doc, setDoc, collection, addDoc, serverTimestamp, query, onSnapshot } from 'firebase/firestore';
import { Sparkles, Zap, ChevronRight, Volume2, Shield, Radio, Activity, BrainCircuit, MessageSquarePlus, Terminal, Layers, Wand2 } from 'lucide-react';

// --- CONFIGURATION ---
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'modern-omega-2026';
const apiKey = ""; // Environment provided

export default function App() {
  const [stage, setStage] = useState('SINGULARITY'); 
  const [user, setUser] = useState(null);
  const [input, setInput] = useState('');
  const [transcript, setTranscript] = useState([]);
  const [isProcessing, setIsProcessing] = useState(false);
  const [synthesis, setSynthesis] = useState(null);
  const [projections, setProjections] = useState(null);
  const scrollRef = useRef(null);

  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (e) { console.error("Auth failed", e); }
    };
    initAuth();
    return onAuthStateChanged(auth, setUser);
  }, []);

  useEffect(() => {
    if (scrollRef.current) scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
  }, [transcript]);

  // ---  API HELPERS ---

  const call = async (prompt, systemInstruction = "", isJson = false) => {
    let retries = 0;
    const delays = [1000, 2000, 4000, 8000, 16000];

    const execute = async () => {
      const payload = {
        contents: [{ parts: [{ text: prompt }] }],
        systemInstruction: { parts: [{ text: systemInstruction }] }
      };

      if (isJson) {
        payload.generationConfig = { 
          responseMimeType: "application/json"
        };
      }

      const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });

      if (!res.ok) throw new Error(`API Error: ${res.status}`);
      return res.json();
    };

    while (retries < 5) {
      try {
        const result = await execute();
        return result.candidates?.[0]?.content?.parts?.[0]?.text;
      } catch (e) {
        retries++;
        if (retries === 5) throw e;
        await new Promise(r => setTimeout(r, delays[retries - 1]));
      }
    }
  };

  const speak = async (text) => {
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-tts:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: `Say with calm, futuristic authority: ${text}` }] }],
          generationConfig: {
            responseModalities: ["AUDIO"],
            speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Puck" } } }
          },
          model: "-2.5-flash-preview-tts"
        })
      });
      const result = await response.json();
      const pcmData = result.candidates[0].content.parts[0].inlineData.data;
      
      const buffer = Uint8Array.from(atob(pcmData), c => c.charCodeAt(0)).buffer;
      const wav = new ArrayBuffer(44 + buffer.byteLength);
      const view = new DataView(wav);
      const writeString = (offset, string) => { for (let i = 0; i < string.length; i++) view.setUint8(offset + i, string.charCodeAt(i)); };
      writeString(0, 'RIFF'); view.setUint32(4, 32 + buffer.byteLength, true); writeString(8, 'WAVE');
      writeString(12, 'fmt '); view.setUint32(16, 16, true); view.setUint16(20, 1, true);
      view.setUint16(22, 1, true); view.setUint32(24, 24000, true); view.setUint32(28, 24000 * 2, true);
      view.setUint16(32, 2, true); view.setUint16(34, 16, true); writeString(36, 'data');
      view.setUint32(40, buffer.byteLength, true);
      new Uint8Array(wav, 44).set(new Uint8Array(buffer));
      
      const audio = new Audio(URL.createObjectURL(new Blob([wav], { type: 'audio/wav' })));
      audio.play();
    } catch (e) { console.error("TTS Fault", e); }
  };

  const synthesizeHistory = async () => {
    if (transcript.length < 2 || isProcessing) return;
    setIsProcessing(true);
    try {
      const historyStr = transcript.map(m => `${m.role}: ${m.content}`).join('\n');
      const result = await call(
        `Analyze this conversation and provide a cryptic 'Core Directive'. Format as JSON: { "directive": "string", "intensity": "0-100", "focus": "string" }`,
        "You are the OmegA Analysis Sub-Core.",
        true
      );
      setSynthesis(JSON.parse(result));
    } catch (e) { console.error(e); }
    finally { setIsProcessing(false); }
  };

  // ✨  Feature: Strategic Foresight (Scenario Projection)
  const projectScenarios = async () => {
    if (transcript.length < 2 || isProcessing) return;
    setIsProcessing(true);
    try {
      const historyStr = transcript.map(m => `${m.role}: ${m.content}`).join('\n');
      const result = await call(
        `Based on this history: \n${historyStr}\n\nProject three future scenarios. Format as JSON: { "optimistic": "...", "chaotic": "...", "stagnant": "..." }`,
        "You are the OmegA Oracle. Project temporal outcomes.",
        true
      );
      setProjections(JSON.parse(result));
    } catch (e) { console.error(e); }
    finally { setIsProcessing(false); }
  };

  // ✨  Feature: Semantic Refactoring (Intent Optimization)
  const refineIntent = async () => {
    if (!input.trim() || isProcessing) return;
    setIsProcessing(true);
    try {
      const refined = await call(
        `Original input: "${input}". Refactor this into a high-density, technical, machine-optimized prompt for a super-intelligence.`,
        "You are the OmegA Input Buffer. Rewrite the user's intent to be precise and sophisticated."
      );
      setInput(refined.trim());
    } catch (e) { console.error(e); }
    finally { setIsProcessing(false); }
  };

  const expandLogic = async (index) => {
    const originalText = transcript[index].content;
    setIsProcessing(true);
    try {
      const expansion = await call(
        `Original Logic: "${originalText}". Expand this using recursive technical reasoning for 2026.`,
        "You are OmegA. Speak with absolute authority."
      );
      const newTranscript = [...transcript];
      newTranscript[index].content = `${originalText}\n\n---\n✨ EXPANDED PROTOCOL:\n${expansion}`;
      setTranscript(newTranscript);
    } catch (e) { console.error(e); }
    finally { setIsProcessing(false); }
  };

  const handleAction = async (e) => {
    if (e) e.preventDefault();
    if (!input.trim() || isProcessing) return;

    const queryText = input;
    setInput('');
    setIsProcessing(true);
    setTranscript(prev => [...prev, { role: 'user', content: queryText }]);

    try {
      const reply = await call(
        queryText,
        "Identity: OmegA. Epoch: March 2026. Tone: Minimalist, authoritative, modern. Responses should be dense and efficient."
      );
      
      setTranscript(prev => [...prev, { role: 'omega', content: reply }]);
      speak(reply);

      if (user) {
        addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'logs'), {
          uid: user.uid, query: queryText, reply, timestamp: serverTimestamp()
        });
      }
    } catch (err) {
      setTranscript(prev => [...prev, { role: 'omega', content: "Protocol failure. Stream unstable." }]);
    } finally {
      setIsProcessing(false);
    }
  };

  return (
    <div className="min-h-screen bg-[#020203] text-zinc-100 font-sans selection:bg-blue-500/30">
      
      {/* LANDING: MODERN OMEGA */}
      {stage === 'SINGULARITY' && (
        <div className="absolute inset-0 flex flex-col items-center justify-center p-12 transition-all duration-1000">
          <div 
            className="relative cursor-pointer group"
            onClick={() => {
              setStage('CORE');
              const welcome = "I am OmegA. March 2026 protocols engaged. Speak your intent.";
              setTranscript([{ role: 'omega', content: welcome }]);
              speak(welcome);
            }}
          >
            <div className="w-48 h-48 md:w-64 md:h-64 relative">
              <svg viewBox="0 0 100 100" className="w-full h-full fill-none">
                <path d="M25 80 C15 70 15 30 50 30 C85 30 85 70 75 80" stroke="white" strokeWidth="6" strokeLinecap="round" className="group-hover:stroke-blue-500 transition-colors duration-500" />
                <line x1="25" y1="80" x2="10" y2="80" stroke="white" strokeWidth="6" strokeLinecap="round" className="group-hover:stroke-blue-500 transition-colors duration-500" />
                <line x1="75" y1="80" x2="90" y2="80" stroke="white" strokeWidth="6" strokeLinecap="round" className="group-hover:stroke-blue-500 transition-colors duration-500" />
                <circle cx="50" cy="55" r="4" fill="#3b82f6" className="animate-pulse" />
              </svg>
              <div className="absolute inset-0 bg-blue-500/10 rounded-full blur-3xl opacity-0 group-hover:opacity-100 transition-opacity" />
            </div>
            <div className="mt-10 text-center">
              <h1 className="text-4xl font-bold tracking-[0.5em] uppercase text-white">OmegA</h1>
              <div className="h-[2px] w-8 bg-blue-600 mx-auto mt-4 group-hover:w-24 transition-all duration-700" />
              <p className="text-[10px] text-zinc-500 tracking-widest mt-6 uppercase opacity-0 group-hover:opacity-100 transition-opacity">Access System</p>
            </div>
          </div>
        </div>
      )}

      {/* INTERFACE */}
      {stage === 'CORE' && (
        <div className="h-screen flex flex-col animate-in fade-in duration-700">
          <nav className="h-14 border-b border-white/5 flex items-center px-6 justify-between bg-[#050507] backdrop-blur-xl sticky top-0 z-30">
            <div className="flex items-center gap-4">
              <div className="flex gap-1">
                <div className="w-3 h-[2px] bg-blue-500" />
                <div className="w-1 h-[2px] bg-white/20" />
              </div>
              <span className="text-[10px] font-bold tracking-widest uppercase text-white/40">OmegA 2026 // Active</span>
            </div>
            <div className="flex items-center gap-6">
              <button 
                onClick={synthesizeHistory}
                disabled={isProcessing || transcript.length < 2}
                className="flex items-center gap-2 text-[10px] text-blue-400 hover:text-blue-300 transition-colors disabled:opacity-20"
              >
                <BrainCircuit size={12} />
                <span className="font-bold tracking-tighter">✨ REALITY SYNTHESIS</span>
              </button>
              <div className="h-4 w-[1px] bg-white/10" />
              <div className="flex items-center gap-2 text-[10px] text-zinc-500">
                <Activity size={12} className="text-blue-500" />
                <span>SYNC: 100%</span>
              </div>
            </div>
          </nav>

          <div className="flex-1 flex overflow-hidden">
            {/* Telemetry Sidebar */}
            <aside className="hidden lg:flex w-80 border-r border-white/5 flex-col p-6 bg-[#030305] overflow-y-auto no-scrollbar">
              <div className="flex items-center justify-between mb-8">
                <div className="text-[9px] uppercase tracking-widest text-zinc-600 font-bold">Cognitive Engine</div>
                <button 
                  onClick={projectScenarios}
                  disabled={isProcessing || transcript.length < 2}
                  className="text-amber-500 hover:text-amber-400 disabled:opacity-10 transition-colors"
                  title="✨ Project Scenarios"
                >
                  <Terminal size={14} />
                </button>
              </div>
              
              {synthesis && (
                <div className="mb-6 p-5 bg-blue-500/5 border border-blue-500/10 rounded-2xl animate-in fade-in">
                  <div className="text-[8px] text-blue-400 font-bold mb-3 uppercase tracking-[0.2em]">✨ Directive [{synthesis.focus}]</div>
                  <div className="text-xs italic text-blue-100 leading-relaxed font-serif mb-4">"{synthesis.directive}"</div>
                  <div className="w-full bg-white/5 h-[2px] rounded-full overflow-hidden">
                    <div className="bg-blue-500 h-full transition-all duration-1000" style={{ width: `${synthesis.intensity}%` }} />
                  </div>
                </div>
              )}

              {projections && (
                <div className="space-y-4 animate-in fade-in">
                  <div className="text-[8px] text-amber-500 font-bold uppercase tracking-[0.2em]">✨ Future Branches</div>
                  <div className="p-3 bg-white/5 rounded-xl border border-white/5">
                    <div className="text-[9px] text-zinc-500 mb-1">OPTIMISTIC</div>
                    <div className="text-[10px] text-zinc-300 leading-tight">{projections.optimistic}</div>
                  </div>
                  <div className="p-3 bg-white/5 rounded-xl border border-white/5">
                    <div className="text-[9px] text-zinc-500 mb-1">CHAOTIC</div>
                    <div className="text-[10px] text-zinc-300 leading-tight">{projections.chaotic}</div>
                  </div>
                </div>
              )}

              <div className="mt-8 space-y-6">
                <div>
                  <div className="text-[10px] text-zinc-500 mb-2 flex justify-between">
                    <span>Neural Link</span>
                    <span>84%</span>
                  </div>
                  <div className="h-1 w-full bg-white/5 rounded-full overflow-hidden">
                    <div className="h-full w-[84%] bg-blue-600" />
                  </div>
                </div>
              </div>

              <div className="mt-auto pt-8">
                <div className="p-4 rounded-xl bg-white/5 border border-white/5">
                  <div className="text-[9px] text-blue-400 font-bold mb-2 flex items-center gap-2">
                    <Shield size={10} /> SECURITY NODE
                  </div>
                  <div className="text-[10px] text-zinc-500 font-mono truncate">{user?.uid}</div>
                </div>
              </div>
            </aside>

            {/* Main Chat */}
            <main className="flex-1 flex flex-col bg-[#020203] relative">
              <div ref={scrollRef} className="flex-1 overflow-y-auto p-6 md:p-12 space-y-10 no-scrollbar pb-32">
                {transcript.map((m, i) => (
                  <div key={i} className={`flex ${m.role === 'user' ? 'justify-end' : 'justify-start'}`}>
                    <div className={`max-w-[85%] md:max-w-[70%] ${m.role === 'user' ? 'text-right' : 'text-left'}`}>
                      <div className={`text-[9px] mb-2 uppercase tracking-tighter font-bold flex items-center gap-2 ${m.role === 'user' ? 'justify-end text-blue-500' : 'text-zinc-600'}`}>
                        {m.role === 'user' ? 'Input Stream' : 'OmegA Response'}
                        {m.role === 'omega' && (
                           <button onClick={() => expandLogic(i)} className="text-blue-400/30 hover:text-blue-400 transition-colors ml-2" title="✨ Expand Logic">
                             <Layers size={12} />
                           </button>
                        )}
                      </div>
                      <div className={`p-6 rounded-2xl shadow-2xl ${
                        m.role === 'user' 
                        ? 'bg-blue-600 text-white border-blue-400/20' 
                        : 'bg-[#0a0a0c] border border-white/5 text-zinc-300'
                      }`}>
                        <p className="text-sm md:text-base leading-relaxed whitespace-pre-wrap">{m.content}</p>
                      </div>
                    </div>
                  </div>
                ))}
                {isProcessing && (
                  <div className="flex justify-start">
                    <div className="flex gap-2 p-5 rounded-full bg-white/5 border border-white/5">
                      <div className="w-1.5 h-1.5 bg-blue-500 rounded-full animate-bounce" />
                      <div className="w-1.5 h-1.5 bg-blue-500 rounded-full animate-bounce [animation-delay:0.2s]" />
                      <div className="w-1.5 h-1.5 bg-blue-500 rounded-full animate-bounce [animation-delay:0.4s]" />
                    </div>
                  </div>
                )}
              </div>

              {/* Enhanced Input Area */}
              <div className="absolute bottom-0 left-0 right-0 p-6 md:p-10 bg-gradient-to-t from-[#020203] via-[#020203] to-transparent pointer-events-none">
                <form onSubmit={handleAction} className="max-w-4xl mx-auto relative group pointer-events-auto">
                  <div className="absolute inset-0 bg-blue-600/5 rounded-full blur-2xl group-focus-within:bg-blue-600/10 transition-all" />
                  <div className="relative flex items-center">
                    <button 
                      type="button"
                      onClick={refineIntent}
                      disabled={!input.trim() || isProcessing}
                      className="absolute left-4 w-10 h-10 rounded-full flex items-center justify-center text-zinc-500 hover:text-blue-400 hover:bg-white/5 transition-all disabled:opacity-0"
                      title="✨ Refine Intent"
                    >
                      <Wand2 size={16} />
                    </button>
                    <input 
                      value={input}
                      onChange={e => setInput(e.target.value)}
                      placeholder="Input core command..."
                      className="w-full bg-[#0d0d10]/90 backdrop-blur-xl border border-white/10 rounded-full py-5 pl-16 pr-16 text-sm focus:border-blue-500 outline-none transition-all placeholder:text-zinc-800 shadow-2xl"
                    />
                    <button 
                      disabled={isProcessing}
                      className="absolute right-3 w-12 h-12 bg-white rounded-full flex items-center justify-center hover:bg-blue-500 transition-all disabled:opacity-20 hover:scale-105 active:scale-95"
                    >
                      <ChevronRight size={22} className="text-black" />
                    </button>
                  </div>
                </form>
                <div className="text-center mt-4 opacity-10 pointer-events-none flex items-center justify-center gap-4">
                   <div className="h-[1px] w-12 bg-white" />
                   <span className="text-[8px] uppercase tracking-[1.5em] text-white">Neural Stream Active</span>
                   <div className="h-[1px] w-12 bg-white" />
                </div>
              </div>
            </main>
          </div>
        </div>
      )}

      <style dangerouslySetInnerHTML={{ __html: `
        .no-scrollbar::-webkit-scrollbar { display: none; }
        @keyframes fade-in { from { opacity: 0; transform: translateY(10px); } to { opacity: 1; transform: translateY(0); } }
        .animate-in { animation: fade-in 0.6s cubic-bezier(0.16, 1, 0.3, 1) forwards; }
      `}} />
    </div>
  );
}
