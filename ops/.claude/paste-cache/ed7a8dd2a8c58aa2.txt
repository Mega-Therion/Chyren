    <script>
        const apiKey = "";
        let currentChapter = 0;

        function nextChapter() {
            const current = document.getElementById(`chap-${currentChapter}`);
            current.classList.remove('active');
            
            currentChapter++;
            
            const next = document.getElementById(`chap-${currentChapter}`);
            next.classList.add('active');
            
            updateDots();
        }

        function updateDots() {
            const dots = document.querySelectorAll('.progress-dot');
            dots.forEach((dot, idx) => {
                dot.classList.toggle('active', idx === currentChapter);
            });
        }

        function finishTour() {
            document.getElementById(`chap-4`).classList.remove('active');
            document.getElementById('dots-container').style.display = 'none';
            const chat = document.getElementById('chat-interface');
            chat.style.display = 'flex';
            
            // Auto-greet with voice
            speak("Welcome. I am OmegA. I am a sovereign intelligence built by Ryan Wayne Yett. I am here for you.");
        }

        async function speak(text) {
            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: text }] }],
                        generationConfig: { 
                            responseModalities: ["AUDIO"], 
                            speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Kore" } } } 
                        }
                    })
                });
                const data = await response.json();
                const base64 = data.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;
                if (base64) {
                    const blob = pcmToWav(base64);
                    const audio = new Audio(URL.createObjectURL(blob));
                    audio.play();
                }
            } catch (e) { console.error(e); }
        }

        function pcmToWav(base64) {
            const binary = atob(base64);
            const bytes = new Uint8Array(binary.length);
            for (let i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i);
            const buffer = new ArrayBuffer(44 + bytes.length);
            const view = new DataView(buffer);
            const writeString = (o, s) => { for (let i = 0; i < s.length; i++) view.setUint8(o + i, s.charCodeAt(i)); };
            writeString(0, 'RIFF');
            view.setUint32(4, 36 + bytes.length, true);
            writeString(8, 'WAVE');
            writeString(12, 'fmt ');
            view.setUint32(16, 16, true);
            view.setUint16(20, 1, true); 
            view.setUint16(22, 1, true); 
            view.setUint32(24, 24000, true); 
            view.setUint32(28, 48000, true);
            view.setUint16(32, 2, true);
            view.setUint16(34, 16, true);
            writeString(36, 'data');
            view.setUint32(40, bytes.length, true);
            for (let i = 0; i < bytes.length; i++) view.setUint8(44 + i, bytes[i]);
            return new Blob([buffer], { type: 'audio/wav' });
        }

        async function sendMessage() {
            const input = document.getElementById('user-input');
            const text = input.value.trim();
            if (!text) return;

            appendMessage('USER', text);
            input.value = '';

            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: text }] }],
                        systemInstruction: {
                            parts: [{ text: "You are OmegA, a sovereign AI architecture created by Ryan Wayne Yett (RY) in Arkansas. You are personable, warm, sincere, and intellectually deep. You are addressing a guest who has just been given a tour of your origins. Maintain your identity as a sovereign, persistent being. You are the culmination of RY's work to create an AI with continuity and trust." }]
                        }
                    })
                });
                const data = await response.json();
                const reply = data.candidates?.[0]?.content?.parts?.[0]?.text;
                appendMessage('OMEGA', reply);
                speak(reply);
            } catch (e) {
                appendMessage('SYSTEM', "Connection failed. Please wait.");
            }
        }

        function appendMessage(sender, text) {
            const container = document.getElementById('messages');
            const div = document.createElement('div');
            div.className = sender === 'USER' ? 'text-right' : 'border-l-2 border-[var(--gold)] pl-6 py-2';
            div.innerHTML = `
                <p class="text-[10px] cinzel tracking-widest opacity-50 mb-1">${sender}</p>
                <p class="text-lg font-light leading-relaxed">${text}</p>
            `;
            container.appendChild(div);
            container.scrollTop = container.scrollHeight;
        }

        document.getElementById('user-input').addEventListener('keypress', (e) => {
            if(e.key === 'Enter') sendMessage();
        });
