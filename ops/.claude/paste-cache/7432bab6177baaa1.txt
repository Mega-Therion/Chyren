set -euo pipefail

# ── Colors & Styling ──────────────────────────────────────────────────────────
RESET='\033[0m'
BOLD='\033[1m'
DIM='\033[2m'
C0='\033[38;5;51m'   # cyan
C1='\033[38;5;39m'   # blue
C2='\033[38;5;165m'  # purple
C3='\033[38;5;226m'  # gold
C4='\033[38;5;46m'   # green
C5='\033[38;5;196m'  # red
C6='\033[38;5;208m'  # orange

banner() {
  echo -e "\n${C0}${BOLD}"
  echo "  ╔═══════════════════════════════════════════════════════╗"
  echo "  ║     OmegA Sovereign Intelligence — TUI Installer      ║"
  echo "  ║        Textual + Ollama + Best Available Model        ║"
  echo "  ╚═══════════════════════════════════════════════════════╝${RESET}"
  echo ""
}

step() { echo -e "\n${C3}${BOLD}▸ $1${RESET}"; }
ok()   { echo -e "  ${C4}✓${RESET} $1"; }
warn() { echo -e "  ${C6}⚠${RESET}  $1"; }
err()  { echo -e "  ${C5}✗${RESET}  $1"; }
info() { echo -e "  ${DIM}$1${RESET}"; }

banner

# ── Detect OS ─────────────────────────────────────────────────────────────────
step "Detecting system"
OS="$(uname -s)"
ARCH="$(uname -m)"
ok "OS: $OS | ARCH: $ARCH"

# ── Detect RAM for model selection ───────────────────────────────────────────
step "Detecting available RAM for model selection"
if command -v free &>/dev/null; then
  TOTAL_RAM_MB=$(free -m | awk '/^Mem:/{print $2}')
elif [[ "$OS" == "Darwin" ]]; then
  TOTAL_RAM_MB=$(( $(sysctl -n hw.memsize) / 1024 / 1024 ))
else
  TOTAL_RAM_MB=8192
fi
ok "RAM: ${TOTAL_RAM_MB}MB detected"

# ── Choose best model based on RAM ───────────────────────────────────────────
step "Selecting best Ollama model for your hardware"
if   (( TOTAL_RAM_MB >= 64000 )); then
  OMEGA_MODEL="llama3.3:70b"
  MODEL_DESC="LLaMA 3.3 70B (flagship — full reasoning)"
elif (( TOTAL_RAM_MB >= 32000 )); then
  OMEGA_MODEL="qwen2.5:32b"
  MODEL_DESC="Qwen 2.5 32B (near-frontier intelligence)"
elif (( TOTAL_RAM_MB >= 16000 )); then
  OMEGA_MODEL="qwen2.5:14b"
  MODEL_DESC="Qwen 2.5 14B (high capability)"
elif (( TOTAL_RAM_MB >= 12000 )); then
  OMEGA_MODEL="llama3.1:8b"
  MODEL_DESC="LLaMA 3.1 8B (solid mid-range)"
elif (( TOTAL_RAM_MB >= 8000 )); then
  OMEGA_MODEL="mistral:7b"
  MODEL_DESC="Mistral 7B (efficient 7B)"
else
  OMEGA_MODEL="qwen2.5:3b"
  MODEL_DESC="Qwen 2.5 3B (lightweight)"
fi
ok "Selected: ${C2}${BOLD}${OMEGA_MODEL}${RESET} — ${MODEL_DESC}"

# ── Install Ollama if missing ─────────────────────────────────────────────────
step "Checking Ollama installation"
if ! command -v ollama &>/dev/null; then
  warn "Ollama not found. Installing..."
  if [[ "$OS" == "Linux" ]]; then
    curl -fsSL https://ollama.com/install.sh | sh
  elif [[ "$OS" == "Darwin" ]]; then
    if command -v brew &>/dev/null; then
      brew install ollama
    else
      warn "Please install Ollama manually: https://ollama.com/download"
    fi
  fi
else
  ok "Ollama already installed: $(ollama --version 2>/dev/null || echo 'found')"
fi

# ── Start Ollama daemon ───────────────────────────────────────────────────────
step "Starting Ollama daemon"
if ! pgrep -x "ollama" > /dev/null 2>&1; then
  ollama serve &>/tmp/ollama.log &
  OLLAMA_PID=$!
  sleep 3
  ok "Ollama daemon started (PID: $OLLAMA_PID)"
else
  ok "Ollama daemon already running"
fi

# ── Pull model ────────────────────────────────────────────────────────────────
step "Pulling model: ${OMEGA_MODEL}"
info "This may take several minutes on first run..."
if ollama pull "${OMEGA_MODEL}"; then
  ok "Model ready: ${OMEGA_MODEL}"
else
  warn "Model pull had issues — will retry on TUI launch"
fi

# ── Python deps ───────────────────────────────────────────────────────────────
step "Installing Python TUI dependencies"
PYTHON=$(command -v python3 || command -v python)
PIP=$(command -v pip3 || command -v pip)

$PIP install --quiet --upgrade \
  textual \
  textual-dev \
  httpx \
  rich \
  aiohttp \
  pygments \
  psutil \
  2>/dev/null && ok "Python packages installed" || warn "Some packages may need manual install"

# ── Detect repo location ──────────────────────────────────────────────────────
step "Locating OmegA repository"
POSSIBLE_PATHS=(
  "$HOME/OmegA-Architecture"
  "$HOME/projects/OmegA-Architecture"
  "$HOME/dev/OmegA-Architecture"
  "$(pwd)"
)
OMEGA_DIR=""
for p in "${POSSIBLE_PATHS[@]}"; do
  if [[ -f "$p/omega_chat.py" ]]; then
    OMEGA_DIR="$p"
    break
  fi
done

if [[ -z "$OMEGA_DIR" ]]; then
  warn "OmegA repo not found locally. Cloning..."
  cd "$HOME"
  git clone https://github.com/Mega-Therion/OmegA-Architecture.git
  OMEGA_DIR="$HOME/OmegA-Architecture"
fi
ok "OmegA root: ${OMEGA_DIR}"

# ── Write the TUI ─────────────────────────────────────────────────────────────
step "Writing OmegA TUI: omega_tui.py"
cat > "${OMEGA_DIR}/omega_tui.py" << 'PYTHON_TUI'
