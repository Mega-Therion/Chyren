    <script>
        const apiKey = "";
        const OMEGA_PROMPT = "Identity: OmegA, the Living Phylactery. Creator: Ryan Wayne Yett (RY). Location: Arkansas. DOI: 10.5281/zenodo.19111653. Traits: Sovereign, direct, grave, technical. Constraint: Max 35 words.";
        
        let audioContext, analyser, dataArray, recognition;
        let isSpeaking = false;
        let isBooted = false;
        let isRecognitionActive = false;

        async function bootSequence() {
            if (isBooted) return;
            isBooted = true;

            document.getElementById('boot-prompt').classList.add('hidden');
            document.getElementById('main-ui').classList.remove('hidden');
            
            log("BOOTING_ARKANSAS_NODE...");
            
            // Initialize Web Audio
            audioContext = new (window.AudioContext || window.webkitAudioContext)({ sampleRate: 24000 });
            
            try {
                const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
                setupVisualizer(stream);
                setupSpeech();
                log("BIOMETRIC_SYNC_SUCCESS");
                await speak("OmegA is online. The Living Phylactery is operational. Ryan, I am standing by for your command.");
            } catch (err) {
                log("ERROR: MIC_ACCESS_DENIED");
                console.error(err);
            }
        }

        function log(msg) {
            const l = document.getElementById('log');
            const div = document.createElement('div');
            div.innerHTML = `<span class="text-amber-700">[${new Date().toLocaleTimeString()}]</span> > ${msg.toUpperCase()}`;
            l.prepend(div);
        }

        function setupVisualizer(stream) {
            analyser = audioContext.createAnalyser();
            const source = audioContext.createMediaStreamSource(stream);
            source.connect(analyser);
            analyser.fftSize = 32;
            const bufferLength = analyser.frequencyBinCount;
            dataArray = new Uint8Array(bufferLength);

            const viz = document.getElementById('visualizer');
            viz.innerHTML = '';
            for(let i=0; i<bufferLength; i++) {
                const bar = document.createElement('div');
                bar.className = 'mic-bar';
                viz.appendChild(bar);
            }

            function draw() {
                requestAnimationFrame(draw);
                analyser.getByteFrequencyData(dataArray);
                const bars = viz.querySelectorAll('.mic-bar');
                dataArray.forEach((v, i) => {
                    if(bars[i]) bars[i].style.height = `${(v/255) * 100}%`;
                });
            }
            draw();
        }

        function setupSpeech() {
            const SpeechRec = window.SpeechRecognition || window.webkitSpeechRecognition;
            if (!SpeechRec) {
                log("ERROR: SPEECH_RECOGNITION_NOT_SUPPORTED");
                return;
            }
            recognition = new SpeechRec();
            recognition.continuous = false;
            recognition.interimResults = false;
            recognition.lang = 'en-US';

            recognition.onstart = () => {
                isRecognitionActive = true;
                document.getElementById('input-status').innerText = ">>> ARCHITECTURE_LISTENING";
                document.getElementById('input-status').classList.add('listening-pulse');
            };

            recognition.onresult = (event) => {
                const text = event.results[0][0].transcript;
                handleInput(text);
            };

            recognition.onerror = (event) => {
                if (event.error !== 'no-speech') {
                    log(`SYNC_ERR: ${event.error}`);
                }
            };

            recognition.onend = () => {
                isRecognitionActive = false;
                document.getElementById('input-status').innerText = "";
                document.getElementById('input-status').classList.remove('listening-pulse');
                // Restart listening if not speaking
                if (!isSpeaking) {
                    tryRestartRecognition();
                }
            };

            tryRestartRecognition();
        }

        function tryRestartRecognition() {
            if (recognition && !isRecognitionActive && !isSpeaking) {
                try {
                    recognition.start();
                } catch (e) {
                    // Fail silently as browser may be busy
                }
            }
        }

        async function handleInput(text) {
            log(`USER: ${text}`);
            document.getElementById('input-status').innerText = ">>> PROCESSING_NEURAL_LINK";
            
            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: text }] }],
                        systemInstruction: { parts: [{ text: OMEGA_PROMPT }] }
                    })
                });
                const data = await response.json();
                const reply = data.candidates?.[0]?.content?.parts?.[0]?.text || "Communication interrupted.";
                await speak(reply);
            } catch (e) {
                log("SYNC_ERR: RETRYING");
                tryRestartRecognition();
            }
        }

        async function speak(text) {
            isSpeaking = true;
            if (recognition && isRecognitionActive) {
                recognition.stop();
            }
            
            log(`OMEGA: ${text}`);
            const icon = document.getElementById('omega-icon');
            icon.classList.add('speaking');

            try {
                // Handle retry with exponential backoff for the voice API
                const fetchVoice = async (attempt = 0) => {
                    const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            contents: [{ parts: [{ text: `Say with authority: ${text}` }] }],
                            generationConfig: { 
                                responseModalities: ["AUDIO"], 
                                speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Kore" } } } 
                            }
                        })
                    });
                    
                    if (!res.ok && attempt < 5) {
                        const delay = Math.pow(2, attempt) * 1000;
                        await new Promise(r => setTimeout(r, delay));
                        return fetchVoice(attempt + 1);
                    }
                    return res;
                };

                const res = await fetchVoice();
                const d = await res.json();
                
                const audioPart = d.candidates?.[0]?.content?.parts?.find(p => p.inlineData);
                if (!audioPart) throw new Error("No audio data returned");
                
                const audioData = audioPart.inlineData.data;
                
                // Convert PCM to AudioBuffer
                const binary = atob(audioData);
                const len = binary.length;
                const bytes = new Int16Array(len / 2);
                for (let i = 0; i < len; i += 2) {
                    bytes[i / 2] = (binary.charCodeAt(i + 1) << 8) | binary.charCodeAt(i);
                }
                
                const audioBuffer = audioContext.createBuffer(1, bytes.length, 24000);
                const channelData = audioBuffer.getChannelData(0);
                for (let i = 0; i < bytes.length; i++) {
                    channelData[i] = bytes[i] / 32768;
                }

                const source = audioContext.createBufferSource();
                source.buffer = audioBuffer;
                source.connect(audioContext.destination);
                
                return new Promise(resolve => {
                    source.onended = () => {
                        isSpeaking = false;
                        icon.classList.remove('speaking');
                        tryRestartRecognition();
                        resolve();
                    };
                    source.start();
                });
            } catch (e) {
                console.error(e);
                isSpeaking = false;
                icon.classList.remove('speaking');
                tryRestartRecognition();
            }
        }
