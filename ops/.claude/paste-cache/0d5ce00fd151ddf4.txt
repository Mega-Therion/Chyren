import { useState, useCallback } from 'react';
import { useOmegaChat } from '@/hooks/use-omega-chat';
import { useVoice } from '@/hooks/use-voice';
import { OmegaHeader } from '@/components/omega/OmegaHeader';
import { ChatView } from '@/components/omega/ChatView';
import { ChatComposer } from '@/components/omega/ChatComposer';
import { ConversationSidebar } from '@/components/omega/ConversationSidebar';
import { TelemetryPanel } from '@/components/omega/TelemetryPanel';
import { VoiceMode } from '@/components/omega/VoiceMode';
import { SystemStateDrawer } from '@/components/omega/SystemStateDrawer';
import { SettingsModal } from '@/components/omega/SettingsModal';
import { ActionSheet } from '@/components/omega/ActionSheet';
import { useIsMobile } from '@/hooks/use-mobile';

const Index = () => {
  const chat = useOmegaChat();
  const isMobile = useIsMobile();

  const [leftOpen, setLeftOpen] = useState(false);
  const [rightOpen, setRightOpen] = useState(false);
  const [settingsOpen, setSettingsOpen] = useState(false);
  const [systemStateOpen, setSystemStateOpen] = useState(false);
  const [actionSheetOpen, setActionSheetOpen] = useState(false);

  const handleVoiceTranscript = useCallback((text: string) => {
    chat.sendMessage(text);
  }, [chat.sendMessage]);

  const voice = useVoice(handleVoiceTranscript);

  const handleAttachNote = useCallback(() => {
    const note = prompt('Enter a text note to attach as context:');
    if (note?.trim()) {
      chat.sendMessage(`[Note attached]: ${note.trim()}`);
    }
  }, [chat.sendMessage]);

  return (
    <div className="h-screen flex omega-cosmic-bg overflow-hidden">
      {/* Left sidebar - conversations */}
      {!isMobile && (
        <ConversationSidebar
          conversations={chat.conversations}
          activeId={chat.activeConvoId}
          onSelect={chat.switchConversation}
          onNew={chat.newConversation}
          onDelete={chat.deleteConversation}
          isOpen={leftOpen}
          onClose={() => setLeftOpen(false)}
        />
      )}
      {isMobile && (
        <ConversationSidebar
          conversations={chat.conversations}
          activeId={chat.activeConvoId}
          onSelect={chat.switchConversation}
          onNew={chat.newConversation}
          onDelete={chat.deleteConversation}
          isOpen={leftOpen}
          onClose={() => setLeftOpen(false)}
        />
      )}

      {/* Main chat area */}
      <div className="flex-1 flex flex-col min-w-0 relative z-10">
        <OmegaHeader
          systemState={chat.systemState}
          onOpenSettings={() => setSettingsOpen(true)}
          onOpenSystemState={() => setSystemStateOpen(true)}
          onToggleLeftPanel={() => setLeftOpen(!leftOpen)}
          onToggleRightPanel={() => setRightOpen(!rightOpen)}
          showPanelToggles
        />

        <ChatView
          messages={chat.activeConvo?.messages || []}
          onRetry={chat.retryMessage}
          voiceSupported={voice.isSupported}
          onEnterVoice={voice.enterVoiceMode}
        />

        <ChatComposer
          onSend={chat.sendMessage}
          onPlusClick={() => setActionSheetOpen(true)}
          onMicClick={() => {
            if (voice.voiceState.mode === 'listening') voice.stopListening();
            else voice.startListening();
          }}
          disabled={chat.isStreaming}
          isListening={voice.voiceState.mode === 'listening'}
        />
      </div>

      {/* Right panel - telemetry */}
      {!isMobile && (
        <TelemetryPanel
          entries={chat.telemetry}
          isOpen={rightOpen}
          onClose={() => setRightOpen(false)}
        />
      )}
      {isMobile && (
        <TelemetryPanel
          entries={chat.telemetry}
          isOpen={rightOpen}
          onClose={() => setRightOpen(false)}
        />
      )}

      {/* Overlays */}
      <VoiceMode
        voiceState={voice.voiceState}
        onStartListening={voice.startListening}
        onStopListening={voice.stopListening}
        onExit={voice.exitVoiceMode}
        onInterrupt={voice.interruptSpeech}
      />

      <SystemStateDrawer
        state={chat.systemState}
        isOpen={systemStateOpen}
        onClose={() => setSystemStateOpen(false)}
      />

      <SettingsModal
        isOpen={settingsOpen}
        onClose={() => setSettingsOpen(false)}
        voiceSettings={voice.voiceState}
        onUpdateVoice={voice.updateSettings}
      />

      <ActionSheet
        isOpen={actionSheetOpen}
        onClose={() => setActionSheetOpen(false)}
        onNewConversation={chat.newConversation}
        onAttachNote={handleAttachNote}
      />
    </div>
  );
};

export default Index;
