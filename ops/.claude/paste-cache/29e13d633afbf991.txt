# 4. Generate Python Explorer
cat << 'EOF' > omega_kg_explorer.py
import json
import argparse
import sys

def load_graph(filepath):
    try:
        with open(filepath, 'r', encoding='utf-8') as f:
            return json.load(f)
    except FileNotFoundError:
        print(f"Error: Could not find {filepath}")
        sys.exit(1)

def main():
    parser = argparse.ArgumentParser(description="OmegA Knowledge Graph Explorer")
    parser.add_argument('query', help="Search term for a node ID or label")
    args = parser.parse_args()

    kg = load_graph('omega_equation_knowledge_graph.json')
    nodes = [n for n in kg['nodes'] if args.query.lower() in n['id'].lower() or args.query.lower() in n['label'].lower()]

    if not nodes:
        print(f"No nodes found matching: {args.query}")
        return

    for node in nodes:
        print("\n" + "="*50)
        print(f" NODE: {node['label']} (ID: {node['id']})")
        print(f" DESC: {node['description']}")
        print("="*50)

if __name__ == "__main__":
    main()
EOF

### What this script does:
1. Prompts you for your username.
2. Creates an `OmegA-Architecture` folder and writes the Epic README (with the rendered diagram), the Math Addendum, and the Python Graph CLI.
3. Pauses and waits for you to literally drag your PDF files and your `omega_equation_knowledge_graph.json` file into the new folder.
4. When you hit Enter, it runs all the `git` commands sequentially to push it directly to your GitHub repo.
