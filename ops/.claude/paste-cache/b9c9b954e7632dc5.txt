    <script>
        const apiKey = "";
        let recognition;
        let isSpeaking = false;
        let currentAudio = null;
        let syncInterval = null;
        let micPausedBySystem = false;

        function startSystem() {
            document.getElementById('init-overlay').style.display = 'none';
            initScene();
            initSpeech();
            setTranscript("Neural feedback loop suppressed.");
        }

        function initSpeech() {
            if (!('webkitSpeechRecognition' in window)) {
                setTranscript("Voice engine error.");
                return;
            }
            recognition = new webkitSpeechRecognition();
            recognition.continuous = true;
            recognition.interimResults = true;
            recognition.lang = 'en-US';

            recognition.onresult = (event) => {
                // Ignore input if system is currently speaking (Gating)
                if (micPausedBySystem) return;

                let interim = "";
                let final = "";

                for (let i = event.resultIndex; i < event.results.length; ++i) {
                    if (event.results[i].isFinal) final += event.results[i][0].transcript;
                    else interim += event.results[i][0].transcript;
                }

                if (interim || final) {
                    const display = interim || final;
                    document.getElementById('user-feedback').innerText = display;
                    document.getElementById('user-feedback').style.color = '#22d3ee';
                }

                if (final.trim().length > 0) {
                    processInput(final);
                }
            };

            recognition.onerror = () => { restartMic(); };
            recognition.onend = () => { if (!micPausedBySystem) restartMic(); };
            recognition.start();
        }

        function restartMic() {
            try { recognition.start(); } catch(e) {}
        }

        function setTranscript(text, animate = false) {
            const container = document.getElementById('omega-transcript');
            if (!animate) {
                container.innerHTML = `<span class="word active" style="opacity: 0.4">${text}</span>`;
            } else {
                container.innerHTML = text.split(/\s+/).map(word => `<span class="word">${word}</span>`).join(" ");
            }
        }

        async function processInput(text) {
            updateUIState("analyzing");
            const response = await getResponse(text);
            if (response) {
                await speak(response);
            }
        }

        async function getResponse(input) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            try {
                const res = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: input }] }],
                        systemInstruction: { parts: [{ text: "You are OmegA. Be concise. NEVER repeat the user back to them. No self-introduction." }] }
                    })
                });
                const data = await res.json();
                return data.candidates?.[0]?.content?.parts?.[0]?.text;
            } catch (e) { return "Link unstable."; }
        }

        async function speak(text) {
            // STEP 1: DEAFEN MIC IMMEDIATELY
            micPausedBySystem = true;
            try { recognition.stop(); } catch(e) {}
            
            isSpeaking = true;
            updateUIState("speaking");
            setTranscript(text, true);

            const words = document.querySelectorAll('.word');
            const url = `https://generativelanguage.googleapis.com/v1beta/models/-2.5-flash-preview-tts:generateContent?key=${apiKey}`;
            
            try {
                const res = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: text }] }],
                        generationConfig: {
                            responseModalities: ["AUDIO"],
                            speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Charon" } } }
                        }
                    })
                });
                const data = await res.json();
                const pcm = data.candidates?.[0]?.content?.parts?.[0]?.inlineData?.data;
                
                const audio = await decodeAndPlay(pcm);
                currentAudio = audio;

                const msPerWord = (audio.duration * 1000) / words.length;
                let idx = 0;
                syncInterval = setInterval(() => {
                    if (idx < words.length) {
                        words[idx].classList.add('active');
                        if (idx > 0) words[idx-1].classList.add('faded');
                        idx++;
                    } else { clearInterval(syncInterval); }
                }, msPerWord);

                audio.onended = () => {
                    endSpeech();
                };
            } catch (e) { endSpeech(); }
        }

        function endSpeech() {
            isSpeaking = false;
            micPausedBySystem = false;
            updateUIState("listening");
            restartMic(); // Re-enable listening only after audio is finished
        }

        function stopVoice() {
            if (currentAudio) { currentAudio.pause(); currentAudio = null; }
            if (syncInterval) clearInterval(syncInterval);
            endSpeech();
        }

        function manualReset() {
            stopVoice();
            document.getElementById('user-feedback').innerText = "System Cleaned";
            setTranscript("Echo-gate active.");
        }

        function updateUIState(state) {
            const badge = document.getElementById('status-text');
            const core = document.getElementById('omega-core');
            const feedback = document.getElementById('user-feedback');

            if (state === "listening") {
                badge.innerText = "Monitoring";
                badge.className = "status-badge listening";
                core.className = "omega-core active";
                feedback.innerText = "Ready";
            } else if (state === "speaking") {
                badge.innerText = "Projecting";
                badge.className = "status-badge speaking";
                core.className = "omega-core muted";
                feedback.innerText = "Mic Gated";
                feedback.style.color = "#444";
            } else if (state === "analyzing") {
                badge.innerText = "Processing";
                core.className = "omega-core active";
            }
        }

        // Standard PCM to WAV for playback
        async function decodeAndPlay(base64) {
            const binary = atob(base64);
            const bytes = new Uint8Array(binary.length);
            for (let i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i);
            const wavHeader = new ArrayBuffer(44);
            const view = new DataView(wavHeader);
            view.setUint32(0, 0x52494646, false); view.setUint32(4, 36 + bytes.length, true);
            view.setUint32(8, 0x57415645, false); view.setUint32(12, 0x666d7420, false);
            view.setUint16(16, 16, true); view.setUint16(20, 1, true);
            view.setUint16(22, 1, true); view.setUint32(24, 24000, true);
            view.setUint32(28, 48000, true); view.setUint16(32, 2, true);
            view.setUint16(34, 16, true); view.setUint32(36, 0x64617461, false);
            view.setUint32(40, bytes.length, true);
            const blob = new Blob([wavHeader, bytes], { type: 'audio/wav' });
            const audio = new Audio(URL.createObjectURL(blob));
            return new Promise(res => { 
                audio.onloadedmetadata = () => { audio.play(); res(audio); };
            });
        }

        function initScene() {
            const scene = new THREE.Scene();
            const camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);
            const renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            document.getElementById('canvas-container').appendChild(renderer.domElement);
            
            const geometry = new THREE.IcosahedronGeometry(2, 1);
            const material = new THREE.MeshBasicMaterial({ color: 0x22d3ee, wireframe: true, transparent: true, opacity: 0.1 });
            const mesh = new THREE.Mesh(geometry, material);
            scene.add(mesh);
            camera.position.z = 5;

            function animate() {
                requestAnimationFrame(animate);
                mesh.rotation.y += isSpeaking ? 0.05 : 0.005;
                mesh.rotation.z += 0.002;
                material.opacity = isSpeaking ? 0.3 : 0.1;
                renderer.render(scene, camera);
            }
            animate();
        }
